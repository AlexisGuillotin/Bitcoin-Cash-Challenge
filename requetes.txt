SELECT `hash`,
  size,
  virtual_size,
  version,
  lock_time,
  block_hash,
  block_number,
  block_timestamp,
  block_timestamp_month,
  input_count,
  output_count,
  input_value,
  output_value,
  is_coinbase,
  fee, output, input FROM `bigquery-public-data.crypto_bitcoin_cash.transactions`,
UNNEST(outputs) AS output,
UNNEST(inputs) AS input
WHERE block_timestamp_month >= "2022-12-24"
